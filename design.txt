//+------------------------------------------------------------------+
//| Calculates the total floating PnL for the current symbol         |
//+------------------------------------------------------------------+
double SanUtils::getTotalTradeProfits(const ulong magicNumber) {
   int totalOrders = OrdersTotal();
   double totalProfit = 0.0;

// Loop backwards to prevent skipped indices
   for(int i = totalOrders - 1; i >= 0; i--) {
      if(
         OrderSelect(i, SELECT_BY_POS, MODE_TRADES)
         &&(OrderMagicNumber() == magicNumber)
      ) {

         // Filter by symbol (ignoring magic number for now as requested)
         if(OrderSymbol() == _Symbol) {

            // Only count active Buy and Sell positions (ignore pending orders)
            if(OrderType() == OP_BUY || OrderType() == OP_SELL) {

               // Sum up net profit: Gross Profit + Swaps + Commissions
               totalProfit += (OrderProfit() + OrderSwap() + OrderCommission());
            }
         }
      }
   }

//return NormalizeDouble(totalProfit,2);
   return (totalOrders>0)?(totalProfit/(double)totalOrders):0.0;
}
